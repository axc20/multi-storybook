# Ramda

<LiveCodeEditor>

```js live=true
() => {
  const { curry, flip, pipe, pathOr } = Ramda;
  const someObj = {
    a: {
      b: {
        c: ['bat', 'cat']
      }
    }
  };
  const add = (a, b) => a + b;
  const subtract = (a, b) => a - b;
  const addTwo = curry(flip(add))(2);
  const subtractFive = curry(flip(subtract))(5);

  return (
    <ul>
      <li>{pathOr('DNE', ['a', 'b', 'd'], someObj)}</li>
      <li>{pathOr('DNE', ['a', 'b', 'c', 1], someObj)}</li>
      <li>{pipe(addTwo, subtractFive)(10)}</li>
    </ul>
  );
};
```
